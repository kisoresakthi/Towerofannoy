def tower_of_annoy(n, source, target, auxiliary):
    if n == 1:
        print(f"Move disk 1 from {source} to {target}")
        return
    tower_of_annoy(n - 1, source, auxiliary, target)
    print(f"Move disk {n} from {source} to {target}")
    tower_of_annoy(n - 1, auxiliary, target, source)

n = int(input("Enter the number of disks: "))
tower_of_annoy(n, 'A', 'C', 'B')
